calendar

Date:{{date}} 

<div id="calendar">
	<span class="date">Date:{{date}}</span>
	<calendar :lang="lang" @change="onChange"></calendar>
</div>
<script src="//cdn.bootcss.com/moment.js/2.17.1/moment.min.js"></script>
<script src="https://unpkg.com/vue/dist/vue.js"></script>
<script src="/js/vue-date-range/vue-date-range.js"></script>
<script>
	new Vue({
		el: '#calendar',
		components: {
			'calendar':daterange.Calendar
		},
		data: {
			lang: 'en',
			date: moment().format('YYYY-MM-DD')
		},
		methods: {
			onChange(date) {
				this.date = date.format('YYYY-MM-DD');
			}
		}
	});
</script>

date-range

Date range:{{range.startDate.format('YYYY-MM-DD') + '~' + range.endDate.format('YYYY-MM-DD')}} 
<div id="range">
	<span class="date">Date range:{{range.startDate.format('YYYY-MM-DD') + '~' + range.endDate.format('YYYY-MM-DD')}}</span>
	<daterange :range="range" :lang="lang" @change="onChange"></daterange>
</div>
<script src="//cdn.bootcss.com/moment.js/2.17.1/moment.min.js"></script>
<script src="https://unpkg.com/vue/dist/vue.js"></script>
<script src="/js/vue-date-range/vue-date-range.js"></script>
<script>
	new Vue({
		el: '#range',
		components: {
			'daterange':daterange.DateRange
		},
		data: {
			lang: 'en',
			range: {
				startDate: moment(),
				endDate: moment().add(7, 'days')
			}
		},
		methods: {
			onChange(range) {
				this.range = range;
			}
		}
	});
</script>
